Volcano Plot

regFxn <- possibly(function(dat){
  lm(log2(IC+0.01) ~ time, data = dat) %>% 
    broom::tidy()
}, otherwise = NULL)

planStats <- plaDataLong %>% 
  nest(statData = -compound) %>% 
  mutate(df = map(statData, regFxn)) %>% 
  unnest_legacy(df) %>% 
  filter(term != "(Intercept)") %>% 
  mutate(p.adj = p.adjust(p.value, method = "BH"))
